package atemscript.compiler.trees;

/** 创建数组实例表达式 */
public class JCNewArray extends JCExpression
{
    /** 数组类型 */
    public JCExpression elemtype;

    /** 数组长度表达式 */
    public JCExpression lengthExpr;

    @Override
    public <D> void scan(TreeScanner<D> v, D arg) { v.visitNewArray(this, arg); }

    @Override
    public <D> JCTree translate(TreeTranslator<D> v, D arg) {
        return v.translateNewArray(this, arg);
    }
}
