package atemscript.compiler.trees;

/** 数组类型, A[] */
public class JCArrayTypeTree extends JCExpression
{
    /** 数组元素类型 */
    public JCExpression elemType;

    @Override
    public <D> void scan(TreeScanner<D> v, D arg) { v.visitArrayType(this, arg); }

    @Override
    public <D> JCTree translate(TreeTranslator<D> v, D arg) {
        return v.translateArrayType(this, arg);
    }
}
